# OpenFlow Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a macOS menu-bar dictation app — a WisprFlow competitor — that uses a hold-to-talk hotkey, runs `tiny-audio-swift` for STT and an MLX-based LLM for cleanup styling, then types the result into the focused app.

**Architecture:** A local Swift package `OpenFlowEngine` owns the entire dictation pipeline behind protocol-based seams (STT, Styler), driven by a `DictationSession` actor with an explicit state machine. A thin AppKit shell (Xcode project) wires the engine to a menu-bar status item, a floating overlay near the caret, a Settings window, and a first-run Setup window.

**Tech Stack:** Swift 6, SwiftPM, AppKit, AVFoundation, ApplicationServices (`AXUIElement`), CoreGraphics (`CGEventTap` / `CGEventPost`), `tiny-audio-swift` (local path dep), `mlx-swift-lm` 2.x (pinned to 2.30.3+ — forced by transitive conflict via `tiny-audio-swift → mlx-audio-swift`; provides `MLXLLM` / `MLXLMCommon`), `swift-huggingface`, `swift-transformers`. Tests use `swift-testing` (`import Testing`).

**Spec:** `docs/superpowers/specs/2026-05-08-openflow-design.md`.

```swift
import Foundation

public enum PipelinePhase: Equatable, Sendable {
    case idle
    case recording
    case transcribing(rawSoFar: String)
    case styling(styledSoFar: String)
    case injecting(text: String)
    case failed(OpenFlowError)
    case cancelled

    public var isTerminal: Bool {
        switch self {
        case .idle, .failed, .cancelled: true
        default: false
        }
    }
}

extension OpenFlowError: Equatable {
    public static func == (lhs: OpenFlowError, rhs: OpenFlowError) -> Bool {
        // identity-based comparison sufficient for state-machine equality;
        // associated Errors are compared by localizedDescription.
        switch (lhs, rhs) {
        case (.microphonePermissionDenied, .microphonePermissionDenied),
             (.accessibilityPermissionMissing, .accessibilityPermissionMissing),
             (.inputMonitoringPermissionMissing, .inputMonitoringPermissionMissing),
             (.stylerTimedOut, .stylerTimedOut),
             (.targetAppLost, .targetAppLost):
            return true
        case (.modelDownloadFailed(let a), .modelDownloadFailed(let b)),
             (.modelLoadFailed(let a), .modelLoadFailed(let b)),
             (.sttFailed(let a), .sttFailed(let b)),
             (.stylerFailed(let a), .stylerFailed(let b)),
             (.audioCaptureFailed(let a), .audioCaptureFailed(let b)):
            return a.localizedDescription == b.localizedDescription
        default:
            return false
        }
    }
}
```

```swift
import Foundation

public actor DictationSession {
    public private(set) var phase: PipelinePhase = .idle

    private let mic: MicCaptureProtocol
    private let transcriber: TranscriberProtocol
    private let styler: StylerProtocol
    private let injector: InjectorProtocol
    private let stylingEnabled: Bool

    private var pipelineTask: Task<Void, Never>?
    private var idleContinuations: [CheckedContinuation<Void, Never>] = []

    public init(
        mic: MicCaptureProtocol,
        transcriber: TranscriberProtocol,
        styler: StylerProtocol,
        injector: InjectorProtocol,
        stylingEnabled: Bool
    ) {
        self.mic = mic
        self.transcriber = transcriber
        self.styler = styler
        self.injector = injector
        self.stylingEnabled = stylingEnabled
    }

    public func press() async {
        guard phase == .idle else { return }
        do {
            try await mic.start()
            phase = .recording
        } catch {
            transitionTerminal(.failed(.audioCaptureFailed(underlying: error)))
        }
    }

    public func release() async {
        guard phase == .recording else { return }
        let samples = await mic.stop()
        phase = .transcribing(rawSoFar: "")
        pipelineTask = Task { [weak self] in
            await self?.runTranscribeStyleInject(samples: samples)
        }
    }

    /// Test helper — completes when phase is terminal (idle/failed/cancelled).
    public func waitForIdle() async {
        if phase.isTerminal { return }
        await withCheckedContinuation { (cont: CheckedContinuation<Void, Never>) in
            idleContinuations.append(cont)
        }
    }

    private func runTranscribeStyleInject(samples: [Float]) async {
        // Phase: transcribing
        var raw = ""
        do {
            for try await delta in transcriber.transcribeStream(samples: samples, sampleRate: 16_000) {
                raw += delta
                phase = .transcribing(rawSoFar: raw)
            }
        } catch {
            transitionTerminal(.failed(.sttFailed(underlying: error)))
            return
        }

        // Phase: styling (or skip if disabled)
        let finalText: String
        if stylingEnabled {
            phase = .styling(styledSoFar: "")
            var styled = ""
            do {
                for try await delta in styler.style(raw) {
                    styled += delta
                    phase = .styling(styledSoFar: styled)
                }
            } catch {
                transitionTerminal(.failed(.stylerFailed(underlying: error)))
                return
            }
            finalText = styled
        } else {
            finalText = raw
        }

        // Phase: injecting
        phase = .injecting(text: finalText)
        do {
            try await injector.insert(finalText)
        } catch {
            transitionTerminal(.failed(.targetAppLost))
            return
        }

        transitionTerminal(.idle)
    }

    private func transitionTerminal(_ newPhase: PipelinePhase) {
        phase = newPhase
        // After surfacing terminal state to observers, snap back to idle if not already.
        // For the state machine, .idle is the resting state; .failed/.cancelled are
        // observed by UI then become .idle on the next press.
        let conts = idleContinuations
        idleContinuations.removeAll()
        for cont in conts { cont.resume() }
    }
}
```

```swift
import AVFoundation
import Foundation

public actor MicCapture: MicCaptureProtocol {
    private let engine = AVAudioEngine()
    private var samples: [Float] = []
    private let targetSampleRate: Double = 16_000
    private var converter: AVAudioConverter?

    public init() {}

    public func start() async throws {
        samples.removeAll(keepingCapacity: true)
        let input = engine.inputNode
        let inputFormat = input.outputFormat(forBus: 0)
        let outputFormat = AVAudioFormat(
            commonFormat: .pcmFormatFloat32,
            sampleRate: targetSampleRate,
            channels: 1,
            interleaved: false
        )!
        let conv = AVAudioConverter(from: inputFormat, to: outputFormat)
        self.converter = conv

        input.installTap(onBus: 0, bufferSize: 4_096, format: inputFormat) { [weak self] buffer, _ in
            guard let self else { return }
            // Convert to 16 kHz mono Float32 outside the actor (real-time thread), then
            // hand off to the actor.
            guard let conv = conv else { return }
            let frameCapacity = AVAudioFrameCount(
                Double(buffer.frameLength) * (outputFormat.sampleRate / inputFormat.sampleRate) + 16
            )
            guard let out = AVAudioPCMBuffer(pcmFormat: outputFormat, frameCapacity: frameCapacity) else { return }
            var error: NSError?
            var fed = false
            conv.convert(to: out, error: &error) { _, status in
                if fed {
                    status.pointee = .endOfStream
                    return nil
                }
                fed = true
                status.pointee = .haveData
                return buffer
            }
            if error != nil { return }
            let frames = Int(out.frameLength)
            guard frames > 0, let chan = out.floatChannelData?[0] else { return }
            let chunk = Array(UnsafeBufferPointer(start: chan, count: frames))
            Task { [weak self] in await self?.append(chunk) }
        }

        engine.prepare()
        do {
            try engine.start()
        } catch {
            input.removeTap(onBus: 0)
            throw error
        }
    }

    public func stop() async -> [Float] {
        engine.inputNode.removeTap(onBus: 0)
        engine.stop()
        let result = samples
        samples.removeAll(keepingCapacity: false)
        return result
    }

    private func append(_ chunk: [Float]) {
        samples.append(contentsOf: chunk)
    }
}
```

```swift
import Foundation

/// Wraps an inner styler with v1 safeguards: timeout, empty-result fallback, length guard.
/// On any safeguard trip, yields the raw transcript as a single chunk and finishes.
public actor SafeguardedStyler: StylerProtocol {
    private let inner: StylerProtocol
    private let timeoutSeconds: Double
    private let lengthMultiple: Double      // e.g. 2.0 => styled cannot exceed 2× raw chars
    private let emptyGuardMinWords: Int     // e.g. 3 => fall back to raw if styled empty AND raw >3 words

    public init(
        inner: StylerProtocol,
        timeoutSeconds: Double = 8.0,
        lengthMultiple: Double = 2.0,
        emptyGuardMinWords: Int = 3
    ) {
        self.inner = inner
        self.timeoutSeconds = timeoutSeconds
        self.lengthMultiple = lengthMultiple
        self.emptyGuardMinWords = emptyGuardMinWords
    }

    public nonisolated func style(_ raw: String) -> AsyncThrowingStream<String, Error> {
        AsyncThrowingStream { continuation in
            Task {
                let timeout = await self.timeoutSeconds
                let lenMul = await self.lengthMultiple
                let minWords = await self.emptyGuardMinWords
                let inner = await self.inner

                let timeoutTask = Task<Void, Never> {
                    try? await Task.sleep(nanoseconds: UInt64(timeout * 1_000_000_000))
                }

                var styled = ""
                let work = Task<Void, Never> {
                    do {
                        for try await delta in inner.style(raw) {
                            if Task.isCancelled { return }
                            styled += delta
                            continuation.yield(delta)
                        }
                    } catch {
                        // Treat inner failure as styling failure — caller decides.
                        continuation.finish(throwing: error)
                        return
                    }

                    // Length guard
                    let cap = max(8, Int(Double(raw.count) * lenMul))
                    if styled.count > cap {
                        // Reset and fall back to raw
                        continuation.yield(__resetMarker)
                        continuation.yield(raw)
                        continuation.finish()
                        return
                    }

                    // Empty guard
                    let trimmed = styled.trimmingCharacters(in: .whitespacesAndNewlines)
                    let rawWords = raw.split(whereSeparator: { $0.isWhitespace }).count
                    if trimmed.isEmpty && rawWords > minWords {
                        continuation.yield(__resetMarker)
                        continuation.yield(raw)
                        continuation.finish()
                        return
                    }

                    continuation.finish()
                }

                // Race: whichever finishes first
                _ = await timeoutTask.value
                if !work.isCancelled {
                    work.cancel()
                    continuation.yield(__resetMarker)
                    continuation.yield(raw)
                    continuation.finish(throwing: OpenFlowError.stylerTimedOut)
                }
            }
        }
    }

    /// Sentinel marker chunk indicating the consumer must clear any accumulated styled text
    /// and replace with the next chunk (the fallback content). The DictationSession uses this
    /// to reset its `styledSoFar` accumulator. Defined as a single-character ASCII control char
    /// (Cancel) which will never appear in real LLM output.
    public static let resetMarker = "\u{0018}"
}

// File-private alias for clarity inside the implementation.
private let __resetMarker = SafeguardedStyler.resetMarker
```

> The reset marker is the engine's mechanism for "throw away what you've shown so far and replace it with this." `DictationSession` will treat this delta as a clear-then-replace signal. We keep this internal to the engine; the UI never sees it.

```swift
import MLXLMCommon

public actor MLXStyler: StylerProtocol {
    public struct Config: Sendable {
        public let modelId: String
        public let maxTokens: Int
        public let temperature: Float
        public init(
            modelId: String = "mlx-community/Qwen3-4B-Instruct-2507-4bit",

            temperature: Float = 0.2
        ) {
            self.modelId = modelId
            self.maxTokens = maxTokens
            self.temperature = temperature
        }

    public typealias ProgressHandler = @Sendable (Double) -> Void

    private let config: Config
    private var container: ModelContainer?

    public init(config: Config = Config()) {
        self.config = config
    }

    /// Download (if needed) and load the model. Safe to call multiple times.
    public func warmUp(progress: ProgressHandler? = nil) async throws {
        if container != nil { return }
        do {
            let modelConfiguration = ModelConfiguration(id: config.modelId)
            self.container = try await LLMModelFactory.shared.loadContainer(
                configuration: modelConfiguration,
                progressHandler: { p in
                    progress?(p.fractionCompleted)
                }
            )
        } catch {
            throw OpenFlowError.modelLoadFailed(underlying: error)
        }
    }

    public nonisolated func style(_ raw: String) -> AsyncThrowingStream<String, Error> {
        AsyncThrowingStream { continuation in
            Task {
                do {
                    try await self.warmUp()
                    guard let container = await self.container else {
                        continuation.finish(throwing: OpenFlowError.modelLoadFailed(
                            underlying: NSError(domain: "OpenFlow", code: -1)))
                        return
                    }
                    let cfg = await self.config
                    let tokenLimit = min(cfg.maxTokens, max(64, Int(Double(raw.count) * 1.5)))

                    let messages: [Chat.Message] = [
                        .system(StylingPrompt.system),
                        .user(StylingPrompt.userMessage(for: raw)),
                    ]
                    let userInput = UserInput(prompt: .chat(messages))
                    let lmInput = try await container.prepare(input: userInput)
                    let stream = try await container.generate(
                        input: lmInput,
                        parameters: GenerateParameters(maxTokens: tokenLimit, temperature: cfg.temperature)
                    )
                    for await event in stream {
                        switch event {
                        case .chunk(let text):
                            continuation.yield(text)
                        case .info, .toolCall:
                            break
                        }
                    }
                    continuation.finish()
                } catch {
                    continuation.finish(throwing: OpenFlowError.stylerFailed(underlying: error))
                }
            }
        }
    }
}
```

```swift
import AppKit
import Foundation
import CoreGraphics

public actor KeyInjector: InjectorProtocol {
    public struct Config: Sendable {
        public var longTextThreshold: Int
        public var keystrokePacingMicros: UInt32
        public init(longTextThreshold: Int = 500, keystrokePacingMicros: UInt32 = 1_000) {
            self.longTextThreshold = longTextThreshold
            self.keystrokePacingMicros = keystrokePacingMicros
        }
    }

    private var config: Config
    private var targetApp: NSRunningApplication?

    public init(config: Config = Config()) {
        self.config = config
    }

    public func setTargetApp(_ app: NSRunningApplication?) {
        self.targetApp = app
    }

    public func updateConfig(_ config: Config) {
        self.config = config
    }

    public nonisolated func insert(_ text: String) async throws {
        guard !text.isEmpty else { return }
        let path = InjectionPath.choose(for: text, longTextThreshold: await self.config.longTextThreshold)
        if let target = await self.targetApp {
            target.activate()
            // Brief delay so the activation takes effect before we post events.
            try? await Task.sleep(nanoseconds: 30_000_000)
        }
        switch path {
        case .typeKeystrokes:
            await self.typeKeystrokes(text)
        case .clipboardPaste:
            await self.clipboardPaste(text)
        }
    }

    private func typeKeystrokes(_ text: String) {
        let pacing = config.keystrokePacingMicros
        for scalar in text.unicodeScalars {
            postUnicode(scalar)
            usleep(pacing)
        }
    }

    private func postUnicode(_ scalar: Unicode.Scalar) {
        let source = CGEventSource(stateID: .combinedSessionState)
        guard let down = CGEvent(keyboardEventSource: source, virtualKey: 0, keyDown: true),
              let up = CGEvent(keyboardEventSource: source, virtualKey: 0, keyDown: false)
        else { return }
        var utf16 = Array(String(scalar).utf16)
        down.keyboardSetUnicodeString(stringLength: utf16.count, unicodeString: &utf16)
        up.keyboardSetUnicodeString(stringLength: utf16.count, unicodeString: &utf16)
        down.post(tap: .cghidEventTap)
        up.post(tap: .cghidEventTap)
    }

    private func clipboardPaste(_ text: String) async {
        let pb = NSPasteboard.general
        let savedItems = pb.pasteboardItems?.compactMap { item -> NSPasteboardItem? in
            let copy = NSPasteboardItem()
            for type in item.types {
                if let data = item.data(forType: type) {
                    copy.setData(data, forType: type)
                }
            }
            return copy
        }
        pb.clearContents()
        pb.setString(text, forType: .string)
        // Post Cmd+V
        postCmdV()
        // Restore clipboard after a brief delay
        try? await Task.sleep(nanoseconds: 200_000_000)
        pb.clearContents()
        if let savedItems { pb.writeObjects(savedItems) }
    }

    private func postCmdV() {
        let source = CGEventSource(stateID: .combinedSessionState)
        let vKey: CGKeyCode = 0x09  // kVK_ANSI_V
        let down = CGEvent(keyboardEventSource: source, virtualKey: vKey, keyDown: true)
        let up = CGEvent(keyboardEventSource: source, virtualKey: vKey, keyDown: false)
        down?.flags = .maskCommand
        up?.flags = .maskCommand
        down?.post(tap: .cghidEventTap)
        up?.post(tap: .cghidEventTap)
    }
}
```

```swift
import Foundation

public enum HotkeyEvent: Equatable, Sendable {
    case pressed
    case released
}

/// Stateful debouncer: feed it raw down/up events with timestamps; receive debounced events.
/// Pure logic — no Foundation timers or threading. Testable.
public actor HotkeyDebouncer {
    private let holdMillis: Int
    private var keyDownAtMillis: Int64?  // when current key-down began
    private var pressedEmitted: Bool = false
    private var debounceTask: Task<Void, Never>?

    public init(holdMillis: Int) {
        self.holdMillis = holdMillis
    }

    /// Feed a raw down event. If sustained `holdMillis`, the continuation will receive `.pressed`.
    /// Internally schedules a delayed emit.
    public func keyDown(at millis: Int64, sink: @Sendable @escaping (HotkeyEvent) -> Void) {
        guard keyDownAtMillis == nil else { return }
        keyDownAtMillis = millis
        pressedEmitted = false
        let delay = UInt64(holdMillis) * 1_000_000
        debounceTask = Task { [weak self] in
            try? await Task.sleep(nanoseconds: delay)
            guard let self else { return }
            await self.fireIfStillDown(sink: sink)
        }
    }

    public func keyUp(at millis: Int64, sink: @Sendable @escaping (HotkeyEvent) -> Void) {
        defer {
            keyDownAtMillis = nil
            pressedEmitted = false
            debounceTask?.cancel()
            debounceTask = nil
        }
        if pressedEmitted {
            sink(.released)
        }
        // If no `pressed` was emitted yet, the hold was below threshold — emit nothing.
    }

    private func fireIfStillDown(sink: @Sendable @escaping (HotkeyEvent) -> Void) {
        if keyDownAtMillis != nil && !pressedEmitted {
            pressedEmitted = true
            sink(.pressed)
        }
    }
}
```

```swift
import AppKit
import CoreGraphics
import Foundation

public final class HotkeyWatcher: @unchecked Sendable {
    private var config: HotkeyConfig
    private let onEvent: @Sendable (HotkeyEvent) -> Void
    private var tap: CFMachPort?
    private var runLoopSource: CFRunLoopSource?
    private var thread: Thread?
    private let debouncer: HotkeyDebouncer

    public init(config: HotkeyConfig, onEvent: @escaping @Sendable (HotkeyEvent) -> Void) {
        self.config = config
        self.onEvent = onEvent
        self.debouncer = HotkeyDebouncer(holdMillis: config.holdDebounceMillis)
    }

    public func updateConfig(_ config: HotkeyConfig) {
        self.config = config
    }

    public func start() throws {
        guard tap == nil else { return }
        let mask = (1 << CGEventType.flagsChanged.rawValue)
                 | (1 << CGEventType.keyDown.rawValue)
                 | (1 << CGEventType.keyUp.rawValue)
        let opaqueSelf = Unmanaged.passUnretained(self).toOpaque()

        guard let newTap = CGEvent.tapCreate(
            tap: .cgSessionEventTap,
            place: .headInsertEventTap,
            options: .listenOnly,
            eventsOfInterest: CGEventMask(mask),
            callback: { _, type, event, refcon in
                guard let refcon else { return Unmanaged.passRetained(event) }
                let watcher = Unmanaged<HotkeyWatcher>.fromOpaque(refcon).takeUnretainedValue()
                watcher.handle(type: type, event: event)
                return Unmanaged.passRetained(event)
            },
            userInfo: opaqueSelf
        ) else {
            throw OpenFlowError.inputMonitoringPermissionMissing
        }
        self.tap = newTap
        self.runLoopSource = CFMachPortCreateRunLoopSource(nil, newTap, 0)

        let t = Thread { [weak self] in
            guard let self else { return }
            CFRunLoopAddSource(CFRunLoopGetCurrent(), self.runLoopSource, .commonModes)
            CGEvent.tapEnable(tap: newTap, enable: true)
            CFRunLoopRun()
        }
        t.name = "OpenFlow.HotkeyWatcher"
        t.start()
        self.thread = t
    }

    public func stop() {
        if let tap { CGEvent.tapEnable(tap: tap, enable: false) }
        if let runLoopSource { CFRunLoopSourceInvalidate(runLoopSource) }
        tap = nil
        runLoopSource = nil
        thread = nil
    }

    private func handle(type: CGEventType, event: CGEvent) {
        let now = Int64(Date().timeIntervalSince1970 * 1000)
        switch (config.isModifier, type) {
        case (true, .flagsChanged):
            let key = CGKeyCode(event.getIntegerValueField(.keyboardEventKeycode))
            guard key == config.virtualKey else { return }
            // Determine pressed/released by current flags state for this modifier.
            // For Right Option (0x3D), the bit is .maskAlternate combined with right-side bit.
            // Simpler: check if the modifier flag is currently set on the event.
            let flags = event.flags
            let isDown = flags.contains(.maskAlternate)  // crude; works for Right-Option default
            Task { [weak self] in
                guard let self else { return }
                if isDown {
                    await self.debouncer.keyDown(at: now, sink: self.onEvent)
                } else {
                    await self.debouncer.keyUp(at: now, sink: self.onEvent)
                }
            }
        case (false, .keyDown):
            let key = CGKeyCode(event.getIntegerValueField(.keyboardEventKeycode))
            guard key == config.virtualKey else { return }
            // require modifiers must match exactly:
            let want = config.requiredModifiers
            let masked = event.flags.intersection([.maskCommand, .maskAlternate, .maskControl, .maskShift])
            guard masked == want else { return }
            Task { [weak self] in
                guard let self else { return }
                await self.debouncer.keyDown(at: now, sink: self.onEvent)
            }
        case (false, .keyUp):
            let key = CGKeyCode(event.getIntegerValueField(.keyboardEventKeycode))
            guard key == config.virtualKey else { return }
            Task { [weak self] in
                guard let self else { return }
                await self.debouncer.keyUp(at: now, sink: self.onEvent)
            }
        default:
            return
        }
    }
}
```

> The Right-Option detection via `flags.contains(.maskAlternate)` distinguishes left vs right Option imperfectly. A robust implementation reads bit 6 of the keyboard mask (NSEvent.ModifierFlags right-side variants). For v1, this default-key path is good enough; left-Option will also trigger. Document this in `App/SMOKE.md` and revisit if it's a pain.

```swift
import AppKit
import Foundation
import OpenFlowEngine

@MainActor
final class AppCoordinator {
    private let store = SettingsStore.shared
    private let statusItem: StatusItemController
    private let overlay: OverlayWindowController
    private let toast: ToastPresenter
    private let setupWindow: SetupWindowController

    private let mic: MicCapture
    private let transcriber: TinyAudioTranscriber
    private let rawStyler: MLXStyler
    private let safeStyler: SafeguardedStyler
    private let injector: KeyInjector
    private let session: DictationSession
    private var watcher: HotkeyWatcher?
    private var phaseObserver: Task<Void, Never>?

    init(
        statusItem: StatusItemController,
        overlay: OverlayWindowController,
        toast: ToastPresenter,
        setupWindow: SetupWindowController
    ) {
        self.statusItem = statusItem
        self.overlay = overlay
        self.toast = toast
        self.setupWindow = setupWindow

        self.mic = MicCapture()
        self.transcriber = TinyAudioTranscriber()
        self.rawStyler = MLXStyler()
        self.safeStyler = SafeguardedStyler(inner: rawStyler)
        self.injector = KeyInjector(config: .init(longTextThreshold: store.longTextThreshold))
        self.session = DictationSession(
            mic: mic,
            transcriber: transcriber,
            styler: safeStyler,
            injector: injector,
            stylingEnabled: store.stylingEnabled
        )
    }

    func start() {
        // Hotkey wiring
        let watcher = HotkeyWatcher(config: store.hotkeyConfig) { [weak self] event in
            Task { @MainActor in
                guard let self else { return }
                if self.statusItem.isPaused { return }
                switch event {
                case .pressed:  await self.session.press()
                case .released: await self.session.release()
                }
            }
        }
        do {
            try watcher.start()
            self.watcher = watcher
        } catch {
            // Likely Accessibility/Input Monitoring missing; setup window will surface it.
            setupWindow.show()
        }

        // Phase observer
        phaseObserver = Task { [weak self] in
            await self?.observePhases()
        }
    }

    private func observePhases() async {
        // Polling-based observer. (DictationSession.phase isn't a stream in v1; the actor's
        // mutations are already serialized, so a 60Hz read loop is acceptable for UI updates.)
        var lastPhase: PipelinePhase = .idle
        while !Task.isCancelled {
            try? await Task.sleep(nanoseconds: 16_000_000)  // ~60Hz
            let phase = await session.phase
            if phase != lastPhase {
                lastPhase = phase
                await render(phase)
            }
        }
    }

    private func render(_ phase: PipelinePhase) {
        switch phase {
        case .idle:
            overlay.hide()
            statusItem.setIcon(.idle)
        case .recording:
            overlay.show(state: .init(phase: .recording))
            statusItem.setIcon(.recording)
        case .transcribing(let raw):
            overlay.show(state: .init(phase: .transcribing(text: raw)))
            statusItem.setIcon(.transcribing)
        case .styling(let styled):
            overlay.show(state: .init(phase: .styling(text: styled)))
            statusItem.setIcon(.styling)
        case .injecting:
            overlay.hide()
            statusItem.setIcon(.idle)
        case .failed(let err):
            overlay.show(state: .init(phase: .error(message: err.localizedDescription ?? "Error")))
            statusItem.setIcon(.error)
            toast.show(err.localizedDescription ?? "Error")
            // Auto-clear after 2s by snapping observer back to idle on next tick
            // (DictationSession leaves phase at .failed until next .press; for UI purposes
            //  we re-render idle on a delay):
            Task { @MainActor [weak self] in
                try? await Task.sleep(nanoseconds: 2_000_000_000)
                self?.overlay.hide()
                self?.statusItem.setIcon(.idle)
            }
        case .cancelled:
            overlay.hide()
            statusItem.setIcon(.idle)
        }
    }
}
```
